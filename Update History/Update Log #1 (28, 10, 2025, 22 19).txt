# Before using classes

import pygame
import sys
import math

pygame.init() # Activate pygame.

# We start by creating the game window.
screen = pygame.display.set_mode((1280, 720))
pygame.display.set_caption("Game")
clock = pygame.time.Clock()

# Other important things to initiate.
game_state = "menu"
main_menu_background = pygame.image.load("assets/main_menu_background.png")
main_menu_background = pygame.transform.scale(main_menu_background, (1280, 720))

while True:

    #Collecting inputs.
    for event in pygame.event.get():

        # For exitting the program.
        if event.type == pygame.QUIT:
            pygame.quit()
            sys.exit()
        
        # Manages the activities of the main menu.
        if game_state == "menu":
            if event.type == pygame.KEYDOWN:

                # For exitting the program.
                if event.key == pygame.K_ESCAPE:
                    pygame.quit()
                    sys.exit()
                
                # For proceeding to the game.
                if event.key == pygame.K_RETURN:
                    game_state = "game"
    
    # Manages the main menu itself.
    if game_state == "menu":
        screen.blit(main_menu_background, (0,0))

    # Manages frames/display.
    pygame.display.flip()
    clock.tick(60)
